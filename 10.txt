package pack;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;
import javax.ejb.Stateless;


@Stateless
public class NewSessionBean {
public Boolean saveReservation(String name, int age, int nop, String checkin, String checkout, String mobile) throws SQLException {
        
        Connection con=DriverManager.getConnection("jdbc:derby://localhost:1527/roomresv","priya","priya");
        
        String sql = "INSERT INTO room(name, age, nop, checkin, checkout, mobile) VALUES (?,?,?,?,?,?)";
        
        
        PreparedStatement stmt=con.prepareStatement(sql);
        stmt.setString(1, name);
        stmt.setInt(2, age);
        stmt.setInt(3, nop);
        stmt.setString(4, checkin);
        stmt.setString(5, checkout);
        stmt.setString(6, mobile);
        
        int rows=stmt.executeUpdate();
        return rows>0;
    }
}

import java.io.IOException;
import java.io.PrintWriter;
import javax.ejb.EJB;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import pack.NewSessionBean;


@WebServlet(urlPatterns = {"/NewServlet"})
public class NewServlet extends HttpServlet {
    @EJB
    private NewSessionBean newSessionBean;

    /**
     * Processes requests for both HTTP <code>GET</code> and <code>POST</code>
     * methods.
     *
     * @param request servlet request
     * @param response servlet response
     * @throws ServletException if a servlet-specific error occurs
     * @throws IOException if an I/O error occurs
     */
    protected void processRequest(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        response.setContentType("text/html;charset=UTF-8");
        try (PrintWriter out = response.getWriter()) {
            /* TODO output your page here. You may use following sample code. */
            out.println("<!DOCTYPE html>");
            out.println("<html>");
            out.println("<head>");
            out.println("<title>Servlet NewServlet</title>");            
            out.println("</head>");
            out.println("<body>");
         
            String name=request.getParameter("name");
            int age=Integer.parseInt(request.getParameter("age"));
            int nop=Integer.parseInt(request.getParameter("nop"));
            String checkin=request.getParameter("checkin");
            String checkout=request.getParameter("checkout");
            String mobile=request.getParameter("mobile");
            
                
            try{
             boolean saved = newSessionBean.saveReservation(name, age, nop, checkin, checkout, mobile);
             if(saved){
                 out.println("<h1>Booking Successful!</h1>");
            out.println("<h3>Booking Details:</h3>");

            out.println("<p><strong>Check-Out Date:</strong> " + checkout + "</p>");
            out.println("<p><strong>Mobile:</strong> " + mobile + "</p>");
             }else{
              out.println("<h1>Booking failed!</h1>");
             }
            }catch(Exception e){
                out.println("<h1>Error: " + e.getMessage() + "</h1>");
            }
